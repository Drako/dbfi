PROJECT (dbfi CXX)
CMAKE_MINIMUM_REQUIRED (VERSION 2.8)

SET (Boost_USE_STATIC_LIBS ON)

IF (CMAKE_COMPILER_IS_GNUCXX OR ("${CMAKE_CXX_COMPIKER_ID}" STREQUAL "Clang"))
	ADD_DEFINITIONS (-std=c++0x)
ENDIF ()

ADD_SUBDIRECTORY (src)
